## 1. Default

The default web component shows playback controls and keeps narration hidden until you opt in.

<FlowPlayerElement controls diagram={FLOW} />

```html
<script src="https://cdn.jsdelivr.net/npm/mermaid-flow-player@latest/mermaid-flow-player.element.js"></script>

<mermaid-flow-player controls>
flowchart LR
  A[Hard] -->|Text| B(Round)
  B --> C{Decision}
  C -->|One| D[Result 1]
  C -->|Two| E[Result 2]
</mermaid-flow-player>
```

## 10. Events and Text-to-Speech

The kitchen-sink demo uses `mfp:step` for spoken narration. The same pattern works with the API hooks or the web component events.

<div class="showcase-grid sl-markdown-no-margin">
  <div class="showcase-card">
    <h3>Sequential TTS</h3>
    <FlowPlayerElement id="showcase-tts-sequential" narration narration-text="Press Play to hear the flow narrated" speed="0.5" diagram={TTS_FLOW} />
  </div>
  <div class="showcase-card">
    <h3>Interactive TTS</h3>
    <FlowPlayerElement id="showcase-tts-interactive" mode="interactive" narration narration-text="Click a dashed node to advance" diagram={TTS_FLOW} />
  </div>
</div>

<script is:inline>{`
  (function () {
    if (typeof window === 'undefined' || !('speechSynthesis' in window)) return;
    var ids = ['showcase-tts-sequential', 'showcase-tts-interactive'];
    function speak(text) {
      if (!text) return;
      window.speechSynthesis.cancel();
      var u = new SpeechSynthesisUtterance(String(text));
      u.rate = 1.1;
      window.speechSynthesis.speak(u);
    }
    function wire(el) {
      if (!el || el.dataset.ttsWired) return;
      el.dataset.ttsWired = '1';
      el.addEventListener('mfp:step', function (e) {
        var d = e.detail || {};
        var step = d.step || {};
        speak(d.narration || step.id);
      });
    }
    function init() { ids.forEach(function (id) { wire(document.getElementById(id)); }); }
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', init);
    } else {
      init();
    }
  })();
`}</script>

```html
<script src="https://cdn.jsdelivr.net/npm/mermaid-flow-player@latest/mermaid-flow-player.element.js"></script>

<mermaid-flow-player id="tts-player" controls narration>
flowchart TD
  A[Request] --> B[Auth]
  B --> C{OK?}
  C -->|yes| D[Process]
  C -->|no| E[Reject]
  D --> F[Confirm]
</mermaid-flow-player>

<script>
  const player = document.getElementById('tts-player');

  player.addEventListener('mfp:step', (event) => {
    const { step, narration } = event.detail;
    const text = narration || step.id;
    if (!text || !('speechSynthesis' in window)) return;

    speechSynthesis.cancel();
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.rate = 1.1;
    speechSynthesis.speak(utterance);
  });
</script>
```

```javascript
import { createFlowPlayer } from 'https://cdn.jsdelivr.net/npm/mermaid-flow-player@latest/index.js';

const player = createFlowPlayer({
  root: document.getElementById('diagram'),
  hooks: {
    onStepStart(step, note) {
      if (!('speechSynthesis' in window)) return;
      const text = note || step.id;
      if (!text) return;

      speechSynthesis.cancel();
      const utterance = new SpeechSynthesisUtterance(text);
      utterance.rate = 1.1;
      speechSynthesis.speak(utterance);
    },
  },
});
```
